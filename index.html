Code example:

Your task is to write a function maskify, which changes all but the last four characters into ‘#’. 

					
					function maskify(numb) {

						if (numb.length > 4) {
							let numbHide = '';
							numb.split('').forEach((item, i) => {
								numbHide = (numb.length - i > 4) ? numbHide += '#' : numbHide += item;
							})
							return numbHide;
						} else {
							return numb;
						}
					
					}
